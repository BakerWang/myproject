Class Ext.form.SearchField

Package:Ext.form
Defined In:core-3.2.js
Class:SearchField
Extends:Object
三态的通用查询控件 
Ext.onReady(function() {
var field = new Ext.form.SearchField();
new Ext.Viewport({
layout : 'border',
items : [{
xtype : "panel",
region : "center",
items : [field]
}]
});
field.on("search", function(c, v, e) {
alert("do my search:" + v)
}, this);
field.on("clear", function(c, v, e) {
alert("do my clear:" + v)
}, this);
});
